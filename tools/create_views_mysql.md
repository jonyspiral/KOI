CREATE OR REPLACE VIEW `cajas_resumen` AS
SELECT
  fecha,
  CASE
    WHEN EXISTS (SELECT 1 FROM caja_saldos_diarios csd WHERE csd.caja_fecha = fecha) THEN 'S'
    ELSE 'N'
  END AS caja_cerrada,
  SUM(cd_importe_1) AS cd_importe_1,
  SUM(cd_importe_2) AS cd_importe_2,
  SUM(cd_importe_3) AS cd_importe_3,
  SUM(cd_importe_4) AS cd_importe_4,
  SUM(cd_importe_5) AS cd_importe_5,
  SUM(cd_importe_6) AS cd_importe_6,
  SUM(oi_importe_1) AS oi_importe_1,
  SUM(oi_importe_2) AS oi_importe_2,
  SUM(oi_importe_3) AS oi_importe_3,
  SUM(oi_importe_4) AS oi_importe_4,
  SUM(oi_importe_5) AS oi_importe_5,
  SUM(oi_importe_6) AS oi_importe_6,
  SUM(db_importe_1) AS db_importe_1,
  SUM(db_importe_2) AS db_importe_2,
  SUM(op_importe_1) AS op_importe_1,
  SUM(op_importe_2) AS op_importe_2,
  SUM(gr_importe_1) AS gr_importe_1,
  SUM(cc_importe_1) AS cc_importe_1,
  SUM(cc_importe_2) AS cc_importe_2,
  SUM(cc_importe_3) AS cc_importe_3,
  SUM(cc_importe_4) AS cc_importe_4,
  SUM(cc_importe_5) AS cc_importe_5,
  SUM(cc_importe_6) AS cc_importe_6,
  SUM(cc_importe_total) AS cc_importe_total
FROM (
  /* Cobranza Deudores (los CD de documentos) */
  SELECT
    fecha_caja AS fecha,
    COALESCE(SUM(do.importe_1), 0) AS cd_importe_1,
    COALESCE(SUM(do.importe_2), 0) AS cd_importe_2,
    COALESCE(SUM(do.importe_3), 0) AS cd_importe_3,
    COALESCE(SUM(do.importe_4), 0) AS cd_importe_4,
    COALESCE(SUM(do.importe_5), 0) AS cd_importe_5,
    COALESCE(SUM(do.importe_6), 0) AS cd_importe_6,
    0 AS oi_importe_1, 0 AS oi_importe_2, 0 AS oi_importe_3, 0 AS oi_importe_4, 0 AS oi_importe_5, 0 AS oi_importe_6,
    0 AS db_importe_1, 0 AS db_importe_2,
    0 AS op_importe_1, 0 AS op_importe_2,
    0 AS gr_importe_1,
    0 AS cc_importe_1, 0 AS cc_importe_2, 0 AS cc_importe_3, 0 AS cc_importe_4, 0 AS cc_importe_5, 0 AS cc_importe_6,
    0 AS cc_importe_total
  FROM documentos do
  WHERE fecha_caja IS NOT NULL
    AND tipo_docum = 'REC'
    AND operacion_tipo = 'CD'
  GROUP BY fecha_caja

  UNION
  /* Otros Ingresos (los OI de documentos) */
  SELECT
    fecha_caja AS fecha,
    0 AS cd_importe_1, 0 AS cd_importe_2, 0 AS cd_importe_3, 0 AS cd_importe_4, 0 AS cd_importe_5, 0 AS cd_importe_6,
    COALESCE(SUM(do2.importe_1), 0) AS oi_importe_1,
    COALESCE(SUM(do2.importe_2), 0) AS oi_importe_2,
    COALESCE(SUM(do2.importe_3), 0) AS oi_importe_3,
    COALESCE(SUM(do2.importe_4), 0) AS oi_importe_4,
    COALESCE(SUM(do2.importe_5), 0) AS oi_importe_5,
    COALESCE(SUM(do2.importe_6), 0) AS oi_importe_6,
    0 AS db_importe_1, 0 AS db_importe_2,
    0 AS op_importe_1, 0 AS op_importe_2,
    0 AS gr_importe_1,
    0 AS cc_importe_1, 0 AS cc_importe_2, 0 AS cc_importe_3, 0 AS cc_importe_4, 0 AS cc_importe_5, 0 AS cc_importe_6,
    0 AS cc_importe_total
  FROM documentos do2
  WHERE fecha_caja IS NOT NULL
    AND tipo_docum = 'REC'
    AND operacion_tipo = 'OI'
  GROUP BY fecha_caja

  UNION
  /* Depósitos Bancarios Efectivo (BD / E en bancos_depositos_cabecera) */
  SELECT
    caja_fecha AS fecha,
    0 AS cd_importe_1, 0 AS cd_importe_2, 0 AS cd_importe_3, 0 AS cd_importe_4, 0 AS cd_importe_5, 0 AS cd_importe_6,
    0 AS oi_importe_1, 0 AS oi_importe_2, 0 AS oi_importe_3, 0 AS oi_importe_4, 0 AS oi_importe_5, 0 AS oi_importe_6,
    COALESCE(SUM(ba1.total_boleta_deposito), 0) AS db_importe_1,
    0 AS db_importe_2,
    0 AS op_importe_1, 0 AS op_importe_2,
    0 AS gr_importe_1,
    0 AS cc_importe_1, 0 AS cc_importe_2, 0 AS cc_importe_3, 0 AS cc_importe_4, 0 AS cc_importe_5, 0 AS cc_importe_6,
    0 AS cc_importe_total
  FROM bancos_depositos_cabecera ba1
  WHERE caja_fecha IS NOT NULL
    AND tipo_docum_u_operacion = 'BD'
    AND tipo_valor = 'E'
  GROUP BY caja_fecha

  UNION
  /* Depósitos Bancarios Cheques (BD / C en bancos_depositos_cabecera) */
  SELECT
    caja_fecha AS fecha,
    0 AS cd_importe_1, 0 AS cd_importe_2, 0 AS cd_importe_3, 0 AS cd_importe_4, 0 AS cd_importe_5, 0 AS cd_importe_6,
    0 AS oi_importe_1, 0 AS oi_importe_2, 0 AS oi_importe_3, 0 AS oi_importe_4, 0 AS oi_importe_5, 0 AS oi_importe_6,
    0 AS db_importe_1,
    COALESCE(SUM(ba2.total_boleta_deposito), 0) AS db_importe_2,
    0 AS op_importe_1, 0 AS op_importe_2,
    0 AS gr_importe_1,
    0 AS cc_importe_1, 0 AS cc_importe_2, 0 AS cc_importe_3, 0 AS cc_importe_4, 0 AS cc_importe_5, 0 AS cc_importe_6,
    0 AS cc_importe_total
  FROM bancos_depositos_cabecera ba2
  WHERE caja_fecha IS NOT NULL
    AND tipo_docum_u_operacion = 'BD'
    AND tipo_valor = 'C'
  GROUP BY caja_fecha

  UNION
  /* Órdenes de pago */
  SELECT
    caja_fecha AS fecha,
    0 AS cd_importe_1, 0 AS cd_importe_2, 0 AS cd_importe_3, 0 AS cd_importe_4, 0 AS cd_importe_5, 0 AS cd_importe_6,
    0 AS oi_importe_1, 0 AS oi_importe_2, 0 AS oi_importe_3, 0 AS oi_importe_4, 0 AS oi_importe_5, 0 AS oi_importe_6,
    0 AS db_importe_1, 0 AS db_importe_2,
    COALESCE(SUM(op.imp_entreg_en_valor_1), 0) AS op_importe_1,
    COALESCE(SUM(op.imp_entreg_en_valor_2), 0) AS op_importe_2,
    0 AS gr_importe_1,
    0 AS cc_importe_1, 0 AS cc_importe_2, 0 AS cc_importe_3, 0 AS cc_importe_4, 0 AS cc_importe_5, 0 AS cc_importe_6,
    0 AS cc_importe_total
  FROM ordenes_de_pago op
  WHERE caja_fecha IS NOT NULL
    AND operacion_tipo = 'OP'
  GROUP BY caja_fecha

  UNION
  /* Gastos por Rendición varios (gastos_rendicion) */
  SELECT
    fecha_rendicion AS fecha,
    0 AS cd_importe_1, 0 AS cd_importe_2, 0 AS cd_importe_3, 0 AS cd_importe_4, 0 AS cd_importe_5, 0 AS cd_importe_6,
    0 AS oi_importe_1, 0 AS oi_importe_2, 0 AS oi_importe_3, 0 AS oi_importe_4, 0 AS oi_importe_5, 0 AS oi_importe_6,
    0 AS db_importe_1, 0 AS db_importe_2,
    0 AS op_importe_1, 0 AS op_importe_2,
    COALESCE(SUM(gr.total_importe), 0) AS gr_importe_1,
    0 AS cc_importe_1, 0 AS cc_importe_2, 0 AS cc_importe_3, 0 AS cc_importe_4, 0 AS cc_importe_5, 0 AS cc_importe_6,
    0 AS cc_importe_total
  FROM gastos_rendicion gr
  WHERE fecha_rendicion IS NOT NULL
  GROUP BY fecha_rendicion

  UNION
  /* Cajas Cerradas */
  SELECT
    caja_fecha AS fecha,
    0 AS cd_importe_1, 0 AS cd_importe_2, 0 AS cd_importe_3, 0 AS cd_importe_4, 0 AS cd_importe_5, 0 AS cd_importe_6,
    0 AS oi_importe_1, 0 AS oi_importe_2, 0 AS oi_importe_3, 0 AS oi_importe_4, 0 AS oi_importe_5, 0 AS oi_importe_6,
    0 AS db_importe_1, 0 AS db_importe_2,
    0 AS op_importe_1, 0 AS op_importe_2,
    0 AS gr_importe_1,
    cc.importe_recibido_en_valor_1 AS cc_importe_1,
    cc.importe_recibido_en_valor_2 AS cc_importe_2,
    cc.importe_recibido_en_valor_3 AS cc_importe_3,
    cc.importe_recibido_en_valor_4 AS cc_importe_4,
    cc.importe_recibido_en_valor_5 AS cc_importe_5,
    cc.importe_recibido_en_valor_6 AS cc_importe_6,
    IFNULL(cc.importe_recibido_en_valor_1, 0)
      + IFNULL(cc.importe_recibido_en_valor_2, 0)
      + IFNULL(cc.importe_recibido_en_valor_3, 0)
      + IFNULL(cc.importe_recibido_en_valor_4, 0)
      + IFNULL(cc.importe_recibido_en_valor_5, 0)
      + IFNULL(cc.importe_recibido_en_valor_6, 0) AS cc_importe_total
  FROM caja_saldos_diarios cc
  WHERE caja_fecha IS NOT NULL
) a
GROUP BY fecha;

SELECT p.cod_prov, p.razon_social, p.cuit, p.imputacion_especifica, i.denominacion, p.anulado, p.observaciones_gestion, a.saldo, total_cheques FROM proveedores_datos AS p LEFT JOIN ( SELECT p.cod_prov, IFNULL(SUM((CASE WHEN a.tipo_docum = 'NDB' OR a.tipo_docum = 'FAC' THEN 1 ELSE -1 END) * a.importe_pendiente), 0) saldo FROM proveedores_datos p LEFT JOIN documento_proveedor_aplicacion_v a ON p.cod_prov = a.cod_proveedor AND a.importe_pendiente > 0 AND a.factura_gastos = 'N' GROUP BY p.cod_prov ) a ON p.cod_prov = a.cod_prov LEFT JOIN plan_cuentas i ON i.cuenta = p.imputacion_especifica LEFT JOIN ( SELECT cod_proveedor, SUM(importe) total_cheques FROM cheque WHERE fecha_vencimiento >= relativeDate(NOW(), 'today', 0) AND anulado = 'N' AND cod_proveedor IS NOT NULL GROUP BY cod_proveedor ) total_cheques ON p.cod_prov = total_cheques.cod_proveedor;

CREATE OR REPLACE VIEW `listado_clientes_v` AS
SELECT c.cod_cli cod_cliente, c.razon_social, c.denom_fantasia, c.cuit, c.cod_vendedor, sc.telefono_1, sc.telefono_2, c.email email_cliente, sc.email email_sucursal, pa.cod_pais, pr.cod_provincia, pr.denom_provincia, l.cod_localidad, l.denom_localidad, sc.calle, sc.numero, sc.piso, sc.oficina_depto FROM Clientes c LEFT OUTER JOIN sucursales_clientes sc ON c.cod_casa_central = sc.cod_suc AND c.cod_cli = sc.cod_cli LEFT OUTER JOIN Paises pa ON pa.cod_pais = sc.cod_pais LEFT OUTER JOIN Provincias pr ON pr.cod_provincia = sc.cod_provincia LEFT OUTER JOIN localidades l ON l.cod_localidad = sc.cod_localidad WHERE c.anulado = 'N';

CREATE OR REPLACE VIEW `promedio_tolerancia_pagados` AS
SELECT
  casiPagas.documento_fecha,
  casiPagas.cod_cli,
  casiPagas.empresa,
  casiPagas.factura_nro,
  casiPagas.letra_factura,
  promedioHijas.promedio
FROM
(
  /* Facturas madre vigentes desde 2011-12-01 con posible filtro por suma de hijas */
  SELECT
    facturasNuevas.documento_fecha,
    facturasNuevas.cod_cli,
    facturasNuevas.empresa,
    facturasNuevas.factura_nro,
    facturasNuevas.letra_factura
  FROM
  (
    SELECT
      d.documento_fecha,
      d.cod_cli,
      d.empresa,
      d.factura_nro,
      d.letra_factura,
      d.total_factura
    FROM docum_clientes_cabecera d
    WHERE d.documento_fecha > DATE '2011-12-01'
      AND (d.anulada = 'N' OR d.anulada IS NULL)
      AND d.tipo_docum IN ('FAC','NDB')
      AND d.grado = 'M'
  ) AS facturasNuevas
  LEFT JOIN docum_suma_hijas hijas
    ON  facturasNuevas.empresa       = hijas.empresa
    AND facturasNuevas.factura_nro   = hijas.factura_nro
    AND facturasNuevas.letra_factura = hijas.letra_factura
  /* Si querés activar la condición de “casi pagas”, descomentá:
     WHERE hijas.sumaHijas > (facturasNuevas.total_factura * 0.95)
  */
) AS casiPagas
LEFT JOIN
(
  /* Promedio de “tolerancia” calculado con hijos (grado H) */
  SELECT
    a1.empresa,
    a1.factura_nro,
    a1.letra_factura,
    SUM(
      TIMESTAMPDIFF(
        DAY,
        a2.documento_fecha,
        CASE a1.docum_cancel_tipo
          WHEN 'NCR' THEN a2.documento_fecha             /* nota de crédito: diferencia 0 */
          WHEN 'REC' THEN a3.fecha_recibo                /* recibo: hasta fecha del recibo */
          ELSE a1.fecha_docum_anulacion                  /* otro: hasta fecha de anulación */
        END
      ) * (a1.total_factura / NULLIF(a2.total_factura, 0))
    ) AS promedio
  FROM docum_clientes_cabecera a1
  LEFT JOIN docum_clientes_cabecera a2
    ON  a1.empresa        = a2.empresa
    AND a1.factura_nro    = a2.factura_nro
    AND a1.letra_factura  = a2.letra_factura
    AND a2.grado          = 'M'
    AND a1.tipo_docum     = a2.tipo_docum
    AND (a2.anulada = 'N' OR a2.anulada IS NULL)
  LEFT JOIN recibo_cabecera a3
    ON  a1.empresa        = a3.empresa
    AND a1.docum_cancel_nro = a3.recibo_nro
    AND (a3.anulado = 'N' OR a3.anulado IS NULL)
  WHERE a1.documento_fecha > DATE '2011-12-01'
    AND a1.grado = 'H'
    AND a1.tipo_docum IN ('FAC','NDB')
  GROUP BY a1.empresa, a1.factura_nro, a1.letra_factura
) AS promedioHijas
  ON  casiPagas.empresa       = promedioHijas.empresa
  AND casiPagas.factura_nro   = promedioHijas.factura_nro
  AND casiPagas.letra_factura = promedioHijas.letra_factura
WHERE promedioHijas.promedio IS NOT NULL;
